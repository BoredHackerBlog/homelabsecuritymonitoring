/etc/filebeat/filebeat.yml needs to be edited as root and it needs to contain the following:
be sure to set the password to be the ingest token



filebeat.inputs:
- type: log
enabled: true
paths:
- /home/*/.*_history
- /root/.*_history

filebeat.modules:
- module: system
syslog:
enabled: true
var.paths: ["/var/log/syslog"]
auth:
enabled: true
var.paths: ["/var/log/auth.log"]
